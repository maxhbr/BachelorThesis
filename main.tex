  \foreach \pos in #1 { \fill[blue,opacity=.2] (-.5,#5) rectangle \pos; }
  %\clip (-.5,#4) rectangle (#3,#5);
  \draw[->] (-.5,0) -- (#3+.7,0);
  \draw[->] (0,#4-.2) -- (0,#5+.2);
  \draw (1,0) -- (1,-.1);
  \draw (0,1) -- (-.1,1);
  \foreach \pos in #1 { \node[draw,circle,inner sep=1.5pt,fill=white] at \pos {}; }
