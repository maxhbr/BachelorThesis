\begin{defn}
  Sei $M$ ein $\C\{x\}$-Modul und $K=\C\{x\}[x^{-1}]$, dann ist die

  \[ M[x^{-1}]:=M\otimes_{\C\{x\}}K \,. \]
\end{defn}
